function normalizeBarcode(raw) {
let s = String(raw||'').trim().replace(/[^0-9()]/g, '');
s = s.replace(/[()]/g, '');
const len = s.length;
if (len === 16 && s.startsWith('01')) return s.slice(3);
if (len === 15 && s.startsWith('01')) return s.slice(2);
if (len === 14 && s.startsWith('0')) return s.slice(1);
return s;
}
function getBarcodeVariants(barcode) {
const norm = normalizeBarcode(barcode);
const v = new Set([barcode, norm]);
if (norm.startsWith('2') && norm.length===13) v.add('88'+norm.slice(3));
if (norm.startsWith('88') && norm.length===13) v.add('2'+norm.slice(3,13));
return [...v];
}

// 상품 조회 (심평원 → 로컬 → 신규)
// ═══════════════════════════════════════
async function lookupBarcode(barcode) {
const variants = getBarcodeVariants(barcode);

// 1. 로컬 재고에서 먼저 찾기
for (const v of variants) {
if (inventory[v]) return { ...inventory[v], barcode: v, source: 'local' };
}

// 2. 시트에서 찾기
if (sheetUrl) {
for (const v of variants) {
try {
const res = await fetch(`${sheetUrl}?action=getProduct&barcode=${encodeURIComponent(v)}`);
const data = await res.json();
if (data.success && data.rows?.length) {
return { barcode: v, name: data.rows[0].name, unitLabel: data.rows[0].unitLabel,
stockBox: data.rows[0].stockBox||0, stockUnit: data.rows[0].stockUnit||0,
memo: data.rows[0].memo||'', rows: data.rows, source: 'sheet' };
}
} catch(e){}
}
}

// 3. 심평원 API
try {
const norm = normalizeBarcode(barcode);
const url = `${HIRA_URL}?serviceKey=${HIRA_KEY}&bar_code=${encodeURIComponent(norm)}&numOfRows=1&pageNo=1&type=json`;
const res = await fetch(url);
const data = await res.json();
const items = data?.body?.items;
if (items?.length > 0) {
const name = items[0].ITEM_NAME || items[0].item_name || '';
if (name) return { barcode: norm, name, unitLabel: '묶음', stockBox: 0, stockUnit: 0, memo: '', source: 'hira' };
}
} catch(e){}

return { barcode: normalizeBarcode(barcode), name: '', unitLabel: '묶음', stockBox: 0, stockUnit: 0, memo: '', source: 'new' };
}

else return;

else if(s.includes('포'))type='포';
return{display:qty>1?`${type}(${qty}개)`:type,qty,raw:s};
}

function adjUnitQtyIdx(m,id,ui,delta){
const l=m==='in'?pendingIn:pendingOut;
const item=l.find(x=>String(x.id)===String(id)); if(!item)return;
const u=item.unitRows[ui]; if(!u)return;
u.qty=Math.max(0,u.qty+delta);
const el=document.getElementById('uqty-'+id+'-'+ui); if(el)el.value=u.qty;
updateConfirmBar(m);
}
function setUnitQtyIdx(m,id,ui,val){
const l=m==='in'?pendingIn:pendingOut;
const item=l.find(x=>String(x.id)===String(id)); if(!item)return;
const u=item.unitRows[ui]; if(!u)return;
u.qty=Math.max(0,parseInt(val)||0); updateConfirmBar(m);
}
function adjExtra(m,id,delta){
const l=m==='in'?pendingIn:pendingOut;
const item=l.find(x=>String(x.id)===String(id)); if(!item)return;
item.extraUnits=Math.max(0,(item.extraUnits||0)+delta);
const el=document.getElementById('extra-'+id); if(el)el.value=item.extraUnits;
updateConfirmBar(m);
}
function setExtra(m,id,val){
const l=m==='in'?pendingIn:pendingOut;
const item=l.find(x=>String(x.id)===String(id)); if(!item)return;
item.extraUnits=Math.max(0,parseInt(val)||0); updateConfirmBar(m);
}
function updateConfirmBar(m){
const list=m==='in'?pendingIn:pendingOut;
document.getElementById('cb-'+m+'-summary').innerHTML=`<strong>${list.length}개 품목</strong> 처리 예정`;
}
function toggleAddPanel(m,id){
const l=m==='in'?pendingIn:pendingOut;
const item=l.find(x=>String(x.id)===String(id)); if(!item)return;
item.showAddPanel=!item.showAddPanel;
document.getElementById('uap-'+id)?.classList.toggle('open',item.showAddPanel);
}
async function addUnit(m,id){
const l=m==='in'?pendingIn:pendingOut;
const item=l.find(x=>String(x.id)===String(id)); if(!item)return;
const input=document.getElementById('uap-input-'+id);
const val=input.value.trim(); if(!val){showToast('단위를 입력하세요','err');return;}
const u=parseUnit(val);
item.unitRows.push({rowIndex:null,unitLabel:val,display:u.display,qty:0,unitQty:u.qty,stockBox:0,stockUnit:0});
item.showAddPanel=false; input.value='';
renderPending(m); showToast(`"${u.display}" 단위 추가됨`,'ok');
}
function updateName(m,id,val){
const l=m==='in'?pendingIn:pendingOut;
const i=l.find(x=>String(x.id)===String(id)); if(i)i.name=val;
}
function removeItem(m,id){
if(m==='in')pendingIn=pendingIn.filter(x=>String(x.id)!==String(id));
else pendingOut=pendingOut.filter(x=>String(x.id)!==String(id));
renderPending(m);
}
function clearAll(m){
if(!confirm('목록을 전체 삭제할까요?'))return;
if(m==='in')pendingIn=[];else pendingOut=[];
renderPending(m);
}

async function commitAll(m){
const list=m==='in'?pendingIn:pendingOut;
if(!list.length)return;
for(const item of list){
if(item.isNew&&!item.name.trim()){showToast('상품명을 입력하세요','err');return;}
if(m==='out'){
for(const u of item.unitRows){if(u.qty>0&&u.stockBox<u.qty){showToast(`⚠ ${item.name}: ${u.display} 재고 부족`,'err');return;}}
if((item.extraUnits||0)>0&&item.stockUnit<item.extraUnits){showToast(`⚠ ${item.name}: 낱개 재고 부족`,'err');return;}
}
}
const btn=document.getElementById('cb-'+m+'-btn');
const spin=document.getElementById('spin-'+m);
const txt=document.getElementById('cb-'+m+'-txt');
btn.disabled=true; spin.style.display='block'; txt.textContent='저장 중...';
const now=new Date(); const records=[];

for(const item of list){
const name=item.name.trim();
for(const u of item.unitRows){
if(u.qty===0)continue;
const newStockBox=m==='in'?u.stockBox+u.qty:u.stockBox-u.qty;
// 로컬 재고 업데이트
if(!inventory[item.barcode]) inventory[item.barcode]={name,unitLabel:u.unitLabel,stockBox:0,stockUnit:0,memo:item.memo||''};
inventory[item.barcode].stockBox=newStockBox;
inventory[item.barcode].name=name;
records.push({rowIndex:u.rowIndex,barcode:item.barcode,name,unitLabel:u.unitLabel,type:m,
qty:u.qty,unitQty:u.unitQty,prevStockBox:u.stockBox,newStockBox,
prevStockUnit:item.stockUnit||0,newStockUnit:item.stockUnit||0,time:now.toISOString()});
}
if((item.extraUnits||0)>0){
const newStockUnit=m==='in'?(item.stockUnit||0)+item.extraUnits:(item.stockUnit||0)-item.extraUnits;
if(inventory[item.barcode]) inventory[item.barcode].stockUnit=newStockUnit;
records.push({rowIndex:item.unitRows[0]?.rowIndex,barcode:item.barcode,name,unitLabel:'낱개',type:m,
qty:item.extraUnits,unitQty:1,prevStockBox:0,newStockBox:0,
prevStockUnit:item.stockUnit||0,newStockUnit,isUnit:true,time:now.toISOString()});
}
histLog.unshift({barcode:item.barcode,name,type:m,
unitRows:item.unitRows.filter(u=>u.qty>0).map(u=>({display:u.display,qty:u.qty})),
extraUnits:item.extraUnits||0,time:now.toISOString()});
if(histLog.length>300)histLog.pop();
}

saveInventory();
buildLocalCache();
localStorage.setItem('pubHistory',JSON.stringify(histLog));

// 시트 반영
let sheetOk=false;
if(sheetUrl){
try{
const res=await fetch(sheetUrl,{method:'POST',headers:{'Content-Type':'text/plain'},body:JSON.stringify({action:'batchUpdate',records})});
const data=await res.json(); sheetOk=data.success;
}catch(e){}
}

btn.disabled=false; spin.style.display='none';
txt.textContent=m==='in'?'✅ 입고 전체 확정':'✅ 출고 전체 확정';
const cleared=[...list];
if(m==='in')pendingIn=[];else pendingOut=[];
renderPending(m);
showToast(`${m==='in'?'📦입고':'📤출고'} ${cleared.length}개 완료${sheetUrl?(sheetOk?' · 시트반영':' · 로컬저장'):''}`, 'ok');
}

function renderHistory(){
const today=new Date().toDateString();
const logs=histLog.filter(r=>new Date(r.time).toDateString()===today);
const el=document.getElementById('hist-list');
document.getElementById('hist-count').textContent=logs.length?`(${logs.length}건)`:'';
if(!logs.length){el.innerHTML=`<div class="empty"><div class="empty-icon">📋</div><div class="empty-text">오늘의 이력이 없습니다</div></div>`;return;}
el.innerHTML=logs.map(r=>{
const t=new Date(r.time).toLocaleTimeString('ko-KR',{hour:'2-digit',minute:'2-digit'});
const sign=r.type==='in'?'+':'-';
const unitStr=(r.unitRows||[]).map(u=>`${sign}${u.qty}${u.display}`).join(', ');
const extraStr=r.extraUnits>0?` ${sign}${r.extraUnits}낱개`:'';
return `<div class="hist-item"><div class="hi-badge ${r.type}">${r.type==='in'?'📦':'📤'}</div>
<div class="hi-body"><div class="hi-name">${escHtml(r.name)}</div><div class="hi-meta">${t}</div></div>
<div class="hi-qty ${r.type}">${unitStr}${extraStr}</div></div>`;
}).join('');
}
